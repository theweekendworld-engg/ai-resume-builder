ADD DETERMINISTIC SIGNALS:

```typescript
function calculateDeterministicScore(resumeData: ResumeData, jobDescription: string) {
    const jdLower = jobDescription.toLowerCase();
    const resumeText = JSON.stringify(resumeData).toLowerCase();
    
    // Extract keywords from JD (simple approach)
    const jdWords = jdLower.split(/\W+/).filter(w => w.length > 3);
    const uniqueJdWords = [...new Set(jdWords)];
    
    // Count matches
    const matchedWords = uniqueJdWords.filter(w => resumeText.includes(w));
    const keywordOverlap = (matchedWords.length / uniqueJdWords.length) * 100;
    
    // Skills match
    const skillsLower = resumeData.skills.map(s => s.toLowerCase());
    const skillMatches = uniqueJdWords.filter(w => 
        skillsLower.some(s => s.includes(w))
    );
    const skillsMatch = (skillMatches.length / Math.min(15, uniqueJdWords.length)) * 100;
    
    // Section completeness
    const hasExperience = resumeData.experience.length > 0;
    const hasProjects = resumeData.projects.length > 0;
    const hasSummary = resumeData.personalInfo.summary.length > 50;
    const hasSkills = resumeData.skills.length >= 5;
    const completeness = [hasExperience, hasProjects, hasSummary, hasSkills]
        .filter(Boolean).length * 25;
    
    return {
        keywordOverlap: Math.round(keywordOverlap),
        skillsMatch: Math.round(skillsMatch),
        completeness: Math.round(completeness),
    };
}
```

BLEND WITH LLM:
- Deterministic weight: 30%
- LLM weight: 70%
- Final score = (deterministic * 0.3) + (llmScore * 0.7)

```typescript
export async function extractKeywords(jobDescription: string): Promise<string[]> {
    if (!jobDescription) return [];
    
    // ... existing prompt ...
    
    const content = response.choices[0].message.content;
    if (!content) return [];
    
    const parseResult = await parseWithRetry(content, KeywordsResponseSchema);
    if (parseResult.success) {
        return parseResult.data.keywords;
    }
    
    return [];
}
```

```typescript
export async function scoreReposForJob(
    repos: GitHubRepo[],
    jobDescription: string
): Promise<Array<GitHubRepo & { relevanceScore: number }>> {
    // Quick relevance scoring without AI
    const jdLower = jobDescription.toLowerCase();
    
    return repos.map(repo => {
        let score = 0;
        
        // Language match
        if (repo.language && jdLower.includes(repo.language.toLowerCase())) {
            score += 30;
        }
        
        // Topic matches
        for (const topic of repo.topics) {
            if (jdLower.includes(topic.toLowerCase())) {
                score += 15;
            }
        }
        
        // Description match
        if (repo.description) {
            const descWords = repo.description.toLowerCase().split(/\W+/);
            const matches = descWords.filter(w => jdLower.includes(w)).length;
            score += Math.min(25, matches * 5);
        }
        
        // Stars boost
        score += Math.min(10, repo.stargazers_count);
        
        return { ...repo, relevanceScore: score };
    })
    .sort((a, b) => b.relevanceScore - a.relevanceScore);
}
```
